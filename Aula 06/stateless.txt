import 'package:flutter/material.dart';
import 'package:stateful/color-slider.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      debugShowCheckedModeBanner: false,
      home: const MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  const MyHomePage({super.key});

  @override
  State<MyHomePage> createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  double tamanho = 300;
  final max = 500;
  final min = 50;
  final incremento = 50;

  final double s = 50;
  final double m = 300;
  final double l = 500;

  double red = 0;
  double blue = 0;
  double green = 0;

  @override
  Widget build(BuildContext context) {
    print('build');
    return Scaffold(
        appBar: AppBar(
          title: Text('Flutter State'),
          actions: [
            IconButton(
              icon: const Icon(Icons.remove),
              onPressed: () {
                setState(() {
                  if (tamanho > min) {
                    tamanho -= incremento;
                  }
                });
              },
            ),
            IconButton(
              icon: const Text('S'),
              onPressed: () {
                setState(() {
                  tamanho = s;
                });
              },
            ),
            IconButton(
              icon: const Text('M'),
              onPressed: () {
                setState(() {
                  tamanho = m;
                });
              },
            ),
            IconButton(
              icon: const Text('L'),
              onPressed: () {
                setState(() {
                  tamanho = l;
                });
              },
            ),
            IconButton(
              icon: const Icon(Icons.add),
              onPressed: () {
                setState(() {
                  if (tamanho < max) {
                    tamanho += incremento;
                  }
                });
              },
            ),
          ],
        ),
        body: Column(
          children: [
            Expanded(
              child: Center(
                child: Icon(
                  Icons.lock_clock,
                  size: tamanho,
                  color: Color.fromARGB(
                      255, red.toInt(), green.toInt(), blue.toInt()),
                ),
              ),
            ),
            Padding(
              padding: const EdgeInsets.symmetric(horizontal: 20, vertical: 10),
              child: ColorSlider(
                cor: Colors.red,
                valor: red,
                fn: (value) {
                  setState(() {
                    red = value;
                  });
                },
              ),
            ),
            Padding(
              padding: const EdgeInsets.symmetric(horizontal: 20, vertical: 10),
                 child: ColorSlider(
                cor: Colors.green,
                valor: green,
                fn: (value) {
                  setState(() {
                    green = value;
                  });
                },
              ),
            ),
            Padding(
              padding: const EdgeInsets.symmetric(horizontal: 20, vertical: 10),
                child: ColorSlider(
                cor: Colors.blue,
                valor: blue,
                fn: (value) {
                  setState(() {
                    blue = value;
                  });
                },
              ),
            ),
          ],
        ));
  }
}
